De esta manera y ordenando cada apartado nos queda el siguiente código:

**AvatarControl.cpp**
```cpp
#include "AvatarControl.h"
#include <AzCore/Serialization/SerializeContext.h>
#include <AzCore/Math/Vector3.h>
#include <ROS2/ROS2Bus.h>
#include <AzFramework/Physics/RigidBodyBus.h>
#include <AzCore/Component/TickBus.h>
#include <AzFramework/Physics/PhysicsScene.h>
#include <AzFramework/Physics/Common/PhysicsSceneQueries.h>
#include <AzCore/Interface/Interface.h>


AvatarControl::AvatarControl(){
    m_avatarControlTopicConfiguration.m_topic="/person/cmd_vel";
    m_avatarControlTopicConfiguration.m_type="geometry_msgs::msg::Twist";
}

void AvatarControl::Reflect(AZ::ReflectContext* context){
    if(auto serializeContext = azrtti_cast<AZ::SerializeContext*>(context)){
        serializeContext->Class<AvatarControl, AZ::Component>()
        ->Version(1)
        ->Field("Move Entity", &AvatarControl::moveEntity)
        ->Field("Avatar Control Topic", &AvatarControl::m_avatarControlTopicConfiguration);

        if (auto editContext = serializeContext->GetEditContext())
        {
            editContext->Class<AvatarControl>("AvatarControl", "Control for avatar")
                ->ClassElement(AZ::Edit::ClassElements::EditorData, "")
                ->Attribute(AZ::Edit::Attributes::Category, "Custom")
                ->Attribute(AZ::Edit::Attributes::AppearsInAddComponentMenu, AZ_CRC("Game"))
                ->DataElement(AZ::Edit::UIHandlers::Default, &AvatarControl::moveEntity,"Move Entity","Entity to move")
                ->DataElement(
                        AZ::Edit::UIHandlers::Default,
                        &AvatarControl::m_avatarControlTopicConfiguration,
                        "Topic for avatar control message",
                        "Configuration for ROS2 topic to recive avatar control message");
        }
    }
}

void AvatarControl::OnTick(float deltaTime, AZ::ScriptTimePoint)
{
    if(m_isMoving){
        m_waitTimer += deltaTime;

        if (m_waitTimer >= m_waitDuration)
        {
            m_waitTimer = 0.0f;
            m_isMoving = false;
        }
    }
}

void AvatarControl::Activate(){
    AZ::TickBus::Handler::BusConnect();
    auto m_node = ROS2::ROS2Interface::Get()->GetNode();
    m_subscription = m_node->create_subscription<geometry_msgs::msg::Twist>(m_avatarControlTopicConfiguration.m_topic.data(), m_avatarControlTopicConfiguration.GetQoS(), std::bind(&AvatarControl::TopicCallBack, this, std::placeholders::_1));
}

void AvatarControl::Deactivate(){
    AZ::TickBus::Handler::BusDisconnect();
    m_subscription.reset();
}

void AvatarControl::TopicCallBack(const geometry_msgs::msg::Twist::SharedPtr msg){
    if(!moveEntity.IsValid())
    {
        AZ_Warning("AvatarControl", false, "Unknown Move Entity");
        return;
    }

    MoveEntity(moveEntity, msg);
}

AZ::Quaternion AvatarControl::ApplyRotation(AZ::Transform currentTransform, float rotationAmount){
    AZ::Quaternion rotation = currentTransform.GetRotation();
    if (AZStd::abs(rotationAmount) > 0.0001f)
    {
        AZ::Quaternion zRotation = AZ::Quaternion::CreateRotationZ(rotationAmount);
        rotation =zRotation * rotation;
    }

    return rotation;
}

AZ::Vector3 AvatarControl::ApplyTargetPosition(AZ::Quaternion currentRotation, float moveForward){
    moveForward=moveForward*10;
    AZ::Vector3 targetPosition = currentRotation.TransformVector(AZ::Vector3(0.0f, moveForward, 0.0f));
    targetPosition.SetZ(0.0f);
    if (targetPosition.GetLengthSq() < 0.0001f)
    {
        targetPosition = AZ::Vector3::CreateZero();
    }

    return targetPosition;
}

AZ::Vector3 AvatarControl::ApplyRaycasting(AZ::EntityId entityMove, AZ::Transform currentTransform, AZ::Vector3 globalDirection){
    AZ::Vector3 currentPos = currentTransform.GetTranslation();
    AZ::Vector3 startPos = currentPos;
    startPos.SetZ(startPos.GetZ() - 1.7f);
    AZ::Vector3 finalPos = startPos + globalDirection;

    auto* sceneInterface = AZ::Interface<AzPhysics::SceneInterface>::Get();

    AzPhysics::RayCastRequest request;
    request.m_start = startPos;
    request.m_direction = globalDirection.GetNormalized();
    request.m_distance = globalDirection.GetLength();

    request.m_filterCallback =[entityMove](const AzPhysics::SimulatedBody* body, const Physics::Shape*)
    {
        if (body && body->GetEntityId() == entityMove) return AzPhysics::SceneQuery::QueryHitType::None;

        return AzPhysics::SceneQuery::QueryHitType::Touch;
    };

    AzPhysics::SceneHandle sceneHandle = sceneInterface->GetSceneHandle(AzPhysics::DefaultPhysicsSceneName);

    AzPhysics::SceneQueryHits result = sceneInterface->QueryScene(sceneHandle, &request);

    if (!result.m_hits.empty())
    {
        const auto& hit = result.m_hits[0];

        float safeDistance = AZStd::max(0.0f, hit.m_distance - 0.1f);

        finalPos = startPos + request.m_direction * safeDistance;
    }

    return finalPos;
}

void AvatarControl::MoveEntity(AZ::EntityId entityMove, const geometry_msgs::msg::Twist::SharedPtr msg){
    if (m_isMoving) return;

    AZ::Transform currentTransform;
    AZ::TransformBus::EventResult(currentTransform, entityMove, &AZ::TransformBus::Events::GetWorldTM);

    float moveForward = msg->linear.x;
    float rotationAmount = msg->angular.z;

    // ROTACIÓN
    AZ::Quaternion currentRotation = ApplyRotation(currentTransform, rotationAmount);
    currentTransform.SetRotation(currentRotation);

    // DIRECCIÓN A POSICIÓN FINAL
    AZ::Vector3 globalDirection = ApplyTargetPosition(currentRotation, moveForward);

    // RAYCASTING
    AZ::Vector3 finalTargetPos = ApplyRaycasting(entityMove, currentTransform, globalDirection);

    // MOVER ENTIDAD A POSICIÓN FINAL
    currentTransform.SetTranslation(finalTargetPos);
    Physics::RigidBodyRequestBus::Event(entityMove, &Physics::RigidBodyRequests::SetKinematicTarget, currentTransform);
    m_isMoving = true;
}
```

## Investigación de implementación de gafas VR en el RADIx

Esto es en lo que más tiempo he estado trabajando. En un principio pensé en tener la cámara de la escena de robot, como debería ser en el ejercicio original y luego cambiar al pov de la persona si se detectaban las gafas.

Al final, decidimos entre yo y mi tutor que lo mejor sería siempre tener la ventana con la cámara de la escena del robot y luego que se despliegue otra con el pov de la si tiene el usuario conectado gafas VR.

El principal problema con esto es que en principio, no puedes tener dos cámaras de escena que funcionen a la vez (o funciona la de robot o el pov de la persona), pero esto al final se puede solucionar con la [Gem Multijugador](https://docs.o3de.org/docs/user-guide/networking/multiplayer/) de O3DE.

De esta manera la idea sería un Cliente Host, que sería el punto de vista del robot y un cliente que se conecta a este Host con el punto de vista de la cámara.

![Diagrama Servicio Host-Cliente](Host-Client_Diagram.png)

Para ello he creado este componente que si detecta que es host usa una cámara y si detecta que es cliente usa otra:

**CameraManager.h**
```cpp
#pragma once

#include <AzCore/Component/Component.h>
#include <AzCore/Component/EntityId.h>
#include <AzCore/Component/TickBus.h>

class CameraManager
    : public AZ::Component
    , public AZ::TickBus::Handler
{
public:
    AZ_COMPONENT(CameraManager,"{e4492198-576b-4d18-996a-f0e9f4ffc199}");
    CameraManager();
    ~CameraManager()=default;

    static void Reflect(AZ::ReflectContext* context);
    void OnTick(float deltaTime, AZ::ScriptTimePoint time) override;

    void Activate() override;
    void Deactivate() override;

private:
    void ManageCameras();
    bool IsHost();

    AZ::EntityId cameraClient;
    AZ::EntityId cameraHost;
};
```

**CameraManager.cpp**
```cpp
#include "CameraManager.h"
#include <AzCore/Component/Component.h>
#include <AzCore/Serialization/SerializeContext.h>
#include <AzCore/Serialization/EditContext.h>
#include <AzCore/Component/TransformBus.h>
#include <AzCore/Interface/Interface.h>
#include <AzCore/Component/TickBus.h>
#include <AzFramework/Components/CameraBus.h>
#include <Multiplayer/IMultiplayer.h>

CameraManager::CameraManager() {}

void CameraManager::Reflect(AZ::ReflectContext* context){
    if (auto serializeContext = azrtti_cast<AZ::SerializeContext*>(context))
    {
        serializeContext->Class<CameraManager, AZ::Component>()
            ->Version(1)
            ->Field("Avatar Entity", &CameraManager::cameraClient)
            ->Field("Robot Entity", &CameraManager::cameraHost);

        if (auto editContext = serializeContext->GetEditContext())
        {
            editContext->Class<CameraManager>("CameraManager", "Manage System VR")
                ->ClassElement(AZ::Edit::ClassElements::EditorData, "")
                ->Attribute(AZ::Edit::Attributes::Category, "Custom")
                ->Attribute(AZ::Edit::Attributes::AppearsInAddComponentMenu, AZ_CRC("Game"))
                ->DataElement(AZ::Edit::UIHandlers::Default, &CameraManager::cameraClient, "cámara VR", "")
                ->DataElement(AZ::Edit::UIHandlers::Default, &CameraManager::cameraHost, "cámara Host", "");
        }
    }
}

void CameraManager::OnTick(float deltaTime, AZ::ScriptTimePoint){
    auto* multiplayer = AZ::Interface<Multiplayer::IMultiplayer>::Get();

    if (!multiplayer)
    {
        AZ_Printf("CameraManager","NO ES MULTIPLAYER");
        return;
    }

    auto type = multiplayer->GetAgentType();

    if (type == Multiplayer::MultiplayerAgentType::Uninitialized)
    {
        AZ_Printf("CameraManager","NO SE HA INICIALIZADO EL TIPO DE USUARIO");
        return;
    }

    ManageCameras();
}

void CameraManager::Activate(){
    AZ::TickBus::Handler::BusConnect();
}

void CameraManager::Deactivate() {
    AZ::TickBus::Handler::BusDisconnect();
}

bool CameraManager::IsHost(){
    auto* multiplayer = AZ::Interface<Multiplayer::IMultiplayer>::Get();
    bool m_isHost = false;

    if (multiplayer){
        m_isHost =(multiplayer->GetAgentType() == Multiplayer::MultiplayerAgentType::ClientServer); //Cliente Host
    }

    return m_isHost;
}

void CameraManager::ManageCameras(){
    if (IsHost())
    {
        Camera::CameraRequestBus::Event(cameraHost, &Camera::CameraRequestBus::Events::MakeActiveView);
    }
    else
    {
        Camera::CameraRequestBus::Event(cameraClient, &Camera::CameraRequestBus::Events::MakeActiveView);
    }
}
```

Con esto y configurando los lanzamientos de los GameLaúncher con host y connect respectivamente conseguimos el objetivo que queremos, pero aún hay un problema, y es que los cambios que se realizan en un lado no afectan al otro, son solo locales.

Entonces lo que he estado investigando es como poder sincronizar las posiciones tanto de la persona como del robot para que ambas instancias vean lo mismo y una primera idea sería copiar las posiciones de las dos entidades del cliente host en el cliente y para ello estoy investigando y probando como crear y usar [auto-components](https://docs.o3de.org/docs/user-guide/networking/multiplayer/autocomponents/) que sirven para poder comunicar entre dos instancias en un mismo servidor.
